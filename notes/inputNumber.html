# 数字输入框 InputNumber


通过鼠标或键盘，输入范围内的数值

# 何时使用

  • 当需要获取标准数值时

# 基本使用

Show Code
<script setup lang="ts">
import { ref, watchEffect } from 'vue'
const value = ref(3)
watchEffect(() => {
  console.log('value:', value.value)
})
</script>
<template>
  <InputNumber v-model:value="value" />
</template>

# 步长为小数

Show Code
<script setup lang="ts">
import { ref, watchEffect } from 'vue'
const value = ref(3)
watchEffect(() => {
  console.log('value:', value.value)
})
</script>
<template>
  <InputNumber :step="0.1" v-model:value="value" />
</template>

# 自定义最大最小值

Show Code
<script setup lang="ts">
import { ref, watchEffect } from 'vue'
const value = ref(3)
watchEffect(() => {
  console.log('value:', value.value)
})
</script>
<template>
  <InputNumber :min="0" :max="10" v-model:value="value" />
</template>
